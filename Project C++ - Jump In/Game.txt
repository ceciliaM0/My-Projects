void deseneazaPiesa(int x, int i, int j)
{
    if(x==1)//iepure alb
    {
        IepureAlb(j*120+100,i*120+100);
    }
    if(x==2)//iepure gri
    {
        IepureGri(j*120+100,i*120+100);
    }
    if(x==3)//iepure negru
    {
        IepureNegru(j*120+100,i*120+100);
    }
    if(x==4)//ciuperca
    {
        ciuperca(j*120+100,i*120+100);
    }
    if(x==5)//vulpe orizontala
    {
        vulpeO(j*120+100,i*120+100);
    }
    if(x==6)//vulpe verticala
    {
        vulpeV(j*120+100,i*120+100);
    }

}
void mutarePiesa(int a[5][5])
{
    int ii, ij, fi, fj, ok, sp1, sp2, i, j, IepuriInGauri=0, IepuriInJoc=0, gata, x;
    gata=0;
    for(i=0; i<=4; i++) //afla cati iepuri sunt in joc
        for(j=0; j<=4; j++)
        {
            if((i==0 && (j==0 || j==4)) || (i==2 && j==2) || (i==4 && (j==0 || j==4)))
            {
                if(a[i][j]==1 || a[i][j]==2 || a[i][j]==3)
                    IepuriInGauri++;
            }
            else if(a[i][j]==1 || a[i][j]==2 || a[i][j]==3)
                IepuriInJoc++;
        }
    IepuriInJoc=IepuriInJoc+IepuriInGauri;

    do
    {
        while (!ismouseclick(WM_LBUTTONDOWN));
        getmouseclick(WM_LBUTTONDOWN, mou.xMouse, mou.yMouse);
        ii=(mou.yMouse-100)/120;
        ij=(mou.xMouse-100)/120;
        clearmouseclick(WM_LBUTTONDOWN);
        if(mou.xMouse>710 && mou.xMouse<800 && mou.yMouse>0 && mou.yMouse<70)//buton inapoi la meniu
        {
            break;
        }
        if(mou.xMouse>0 && mou.xMouse<90 && mou.yMouse>0 && mou.yMouse<75)//buton ajutor
        {
            break;
        }
        if(mou.xMouse>0 && mou.xMouse<95 && mou.yMouse>725 && mou.yMouse<800)//buton resetare nivel
        {
            break;
        }

        if(a[ii][ij]==1||a[ii][ij]==2||a[ii][ij]==3)//iepurii
        {
            readimagefile("Patratverde.jpg", ij*120+100, ii*120+100, ij*120+100+120-1, ii*120+100+120-1);
        }
        if(a[ii][ij]==5)//vulpe orizontala
        {
            readimagefile("Patratverde.jpg", ij*120+100, ii*120+100, ij*120+100+120-1, ii*120+100+120-1);
            if(a[ii][ij+1]==5)
            {
                readimagefile("Patratverde.jpg", (ij+1)*120+100, ii*120+100, (ij+1)*120+100+120-1, ii*120+100+120-1);
            }
            else if(a[ii][ij-1]==5)
            {
                readimagefile("Patratverde.jpg", (ij-1)*120+100, ii*120+100, (ij-1)*120+100+120-1, ii*120+100+120-1);
            }
        }
        if(a[ii][ij]==6)//vulpe verticala
        {
            readimagefile("Patratverde.jpg", ij*120+100, ii*120+100, ij*120+100+120-1, ii*120+100+120-1);
            if(a[ii+1][ij]==6)
            {
                readimagefile("Patratverde.jpg", ij*120+100, (ii+1)*120+100, ij*120+100+120-1, (ii+1)*120+100+120-1);
            }
            else if(a[ii-1][ij]==6)
            {
                readimagefile("Patratverde.jpg", ij*120+100, (ii-1)*120+100, ij*120+100+120-1, (ii-1)*120+100+120-1);
            }
        }
        if((ii==0 && (ij==0 || ij==4)) || (ii==2 && ij==2) || (ii==4 && (ij==0 || ij==4)))//gaura
        {
            readimagefile("gaura.jpg", ij*120+100, ii*120+100+1, ij*120+100+120-1, ii*120+100+120);
        }

        while (!ismouseclick(WM_LBUTTONDOWN));
        getmouseclick(WM_LBUTTONDOWN, mou.xMouse, mou.yMouse);
        fi=(mou.yMouse-100)/120;
        fj=(mou.xMouse-100)/120;

        if(mou.xMouse>710 && mou.xMouse<800 && mou.yMouse>0 && mou.yMouse<70)//buton inapoi la meniu
        {
            break;
        }
        if(mou.xMouse>0 && mou.xMouse<90 && mou.yMouse>0 && mou.yMouse<75)//buton ajutor
        {
            break;
        }
        if(mou.xMouse>0 && mou.xMouse<95 && mou.yMouse>725 && mou.yMouse<800)//buton resetare nivel
        {
            break;
        }

        if(ii==fi && ij==fj)
            deseneazaPiesa(a[ii][ij],ii,ij);
        else//mutare iepuri
        {
            if(a[fi][fj]!=0)//daca mutarea finala nu e un spatiu gol, atunci nu ma intereseaza
                ok=0;
            if(a[ii][ij]==1 || a[ii][ij]==2 || a[ii][ij]==3)
            {
                ok=1;
                if(ii!=fi && ij!=fj)//?
                    ok=0;
                if(a[fi][fj]!=0)
                    ok=0;
                if(ii==fi)//mutarea pe linie
                {
                    if(fj-ij==1 || ij-fj==1)//iepurii nu pot sa mearga pe spatiul direct de langa ei
                        ok=0;
                    if(fj<ij)
                    {
                        sp1=fj;
                        sp2=ij;
                    }
                    else
                    {
                        sp1=ij;
                        sp2=fj;
                    }
                    for(j=sp1+1; j<sp2; j++)
                        if(a[ii][j]==0)
                            ok=0;
                }
                if(ij==fj)//mutarea pe coloana
                {
                    if(fi-ii==1 || ii-fi==1)
                        ok=0;
                    if(fi<ii)
                    {
                        sp1=fi;
                        sp2=ii;
                    }
                    else
                    {
                        sp1=ii;
                        sp2=fi;
                    }
                    for(i=sp1+1; i<sp2; i++)
                        if(a[i][ij]==0)
                            ok=0;
                }
                if(ok==0)
                    deseneazaPiesa(a[ii][ij],ii,ij);
                else
                {
                    if(a[ii][ij]!=0 && ((ii==0 && (ij==0 || ij==4)) || (ii==2 && ij==2) || (ii==4 && (ij==0 || ij==4))))//daca pozitia initiala era o gaura
                    {
                        IepuriInGauri--;
                    }
                    if(a[fi][fj]==0 && ((fi==0 && (fj==0 || fj==4)) || (fi==2 && fj==2) || (fi==4 && (fj==0 || fj==4))))//daca pozitia finala e o gaura
                    {
                        IepuriInGauri++;
                        a[fi][fj]=a[ii][ij];
                        a[ii][ij]=0;
                        //cout<<IepuriInGauri<<" ";
                    }
                    if(a[fi][fj]==0)//daca pozitia finala e un spatiu gol oarecare
                    {
                        a[fi][fj]=a[ii][ij];
                        a[ii][ij]=0;
                    }

                    deseneazaPiesa(a[fi][fj],fi,fj);
                    if(IepuriInGauri==IepuriInJoc)
                        gata=1;
                }
            }

        }

        if(a[ii][ij]==5)//mutare vulpe orizontala
        {
            ok=1;
            if(ii!=fi)//nu poti schimba linia
                ok=0;
            if(fj==5)
                ok=0;
            if(a[fi][fj]!=0 && a[fi][fj]!=5 || a[fi][fj+1]!=0 && a[fi][fj+1]!=5)
                ok=0;
            if(fj==4)
                ok=0;
            if(fi<ii)
            {
                sp1=fi;
                sp2=ii;
            }
            else
            {
                sp1=ii;
                sp2=fi;
            }
            for(j=sp1; j<sp2; j++)
                if(a[fi][j]!=0 && a[fi][j]!=5)
                    ok=0;
            if(ok==0)
                deseneazaPiesa(a[ii][ij],ii,ij);
            else
            {
                a[ii][ij]=0;
                if(a[ii][ij+1]==5)
                    a[ii][ij+1]=0;
                else if(a[ii][ij-1]==5)
                    a[ii][ij-1]=0;
                a[fi][fj]=5;
                a[fi][fj+1]=5;
                deseneazaPiesa(a[fi][fj],fi,fj);
            }
        }

        if(a[ii][ij]==6)//mutare vulpe verticala
        {
            ok=1;
            if(ij!=fj)
                ok=0;
            if(a[fi][fj]!=0 && a[fi][fj]!=6 || a[fi+1][fj]!=0 && a[fi+1][fj]!=6)
                ok=0;
            if(fi==4)
                ok=0;
            if(fi<ii)
            {
                sp1=fi;
                sp2=ii;
            }
            else
            {
                sp1=ii;
                sp2=fi;
            }
            for(i=sp1; i<sp2; i++)
                if(a[i][ij]!=0 && a[i][ij]!=6)
                    ok=0;
            if(ok==0)
                deseneazaPiesa(a[ii][ij],ii,ij);
            else
            {
                a[ii][ij]=0;
                if(a[ii+1][ij]==6)
                    a[ii+1][ij]=0;
                else if(a[ii-1][ij]==6)
                    a[ii-1][ij]=0;
                a[fi][fj]=6;
                a[fi+1][fj]=6;
                deseneazaPiesa(a[fi][fj],fi,fj);
            }
        }

    }
    while(gata==0);
}
void pregatesteTabla(int x)
{
    char nivel[52];
    int i, j, a[5][5], k, l, z;
    FILE *ftext2;
    ftext2 = fopen("C:\\Users\\User\\Desktop\\ProiectIP\\Nivele.txt","r");
    k=0;
    while(!feof(ftext2))
    {
        if(fgets(nivel,51,ftext2)&&k!=x)
        {
            k++;
        }
        else if(k==x)
        {
            l=0;
            for(i=0; i<=4; i++)
                for(j=0; j<=4; j++)
                {
                    a[i][j]=nivel[l]-'0';
                    l=l+2;
                }
            for(i=0; i<=4; i++)
                for(j=0; j<=4; j++)
                {
                    z=a[i][j];
                    if(a[i][j]==5 && a[i][j+1]==5)
                        deseneazaPiesa(z,i,j);
                    else if(a[i][j]==6 && a[i+1][j]==6)
                        deseneazaPiesa(z,i,j);
                    else if(z==1 || z==2 || z==3 || z==4)
                        deseneazaPiesa(z,i,j);
                }
            break;
        }
    }
    fclose(ftext2);
    mutarePiesa(a);
}
